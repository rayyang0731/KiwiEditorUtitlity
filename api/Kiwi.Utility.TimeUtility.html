public static string FormatDateTime(DateTime dateTime, string format = "yyyy-MM-dd HH:mm:ss")

Parameters

dateTime DateTime

时间日期

format string

格式,参考 example

Returns

string

格式化后的时间字符串

Examples

格式示例 :

"s" = 2008-04-18T06:30:00 (sortable)

"d" = 04/18/2008 (short date)

"g" = 04/18/2008 06:30 (general short)

"f" = Friday, 18 April 2008 06:30 (full date short)

"t" = 2008-04-18T06:30:00 (short time)

"u" = 2008-4-18 06:30:00Z (universal sortable)

"D" = Friday, 18 April 2008 (long date)

"F" = Friday, 18 April 2008 06:30:00 (full date long)

"G" = 04/18/2008 06:30:00 (general long)

"M" = April 18 (month)

"O" = 2008-04-18T06:30:00.0000000 (ISO 8601)

"R" = Fri, 18 Apr 2008 06:30:00 GMT (RFC 1123)

"T" = 06:30:00(long time)

"U" = Thursday, 17 April 2008 22:30:00(universal full)

"Y" = 2008 April (year month)

"dddd" = Friday (custom)

"M/d/yy" = 4/18/08 (custom)

"h:mm:ss tt zz" = 4:03:05 PM -07 (custom)

"hh:mm:ss t z" = 04:03:05 P -7 (custom)

"yyyy-M-d dddd" = 2008-4-18 Friday (custom)

"HH:m:s zzz" = 16:3:5 -07:00 (custom)

"HH:mm:ss zz" = 16:03:05 -07 (custom)

"MM/dd/yyyy" = 04/18/2008 (custom)

"yy-MM-dd" = 08-04-18 (custom)

"yy-MMM-dd ddd" = 08-Apr-18 Fri (custom)

"yyyy MMMM dd" = 2008 April 18 (custom)
